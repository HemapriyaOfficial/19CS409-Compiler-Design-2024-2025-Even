## PROGRAM
```

#include <stdio.h>
#include <ctype.h>
#include <string.h>

char expr[100];
int pos = 0;

int E(); // Expression
int F(); // Factor

int F() {
    if (isdigit(expr[pos])) {
        while (isdigit(expr[pos])) pos++;
        return 1;
    } else if (isalpha(expr[pos])) {
        while (isalnum(expr[pos])) pos++;
        return 1;
    } else if (expr[pos] == '(') {
        pos++;
        if (E() && expr[pos] == ')') {
            pos++;
            return 1;
        }
        return 0;
    }
    return 0;
}

int E() {
    if (!F()) return 0;
    while (expr[pos] == '+' || expr[pos] == '-' || expr[pos] == '*' || expr[pos] == '/') {
        pos++;
        if (!F()) return 0;
    }
    return 1;
}

int main() {
    printf("Enter expression: ");
    fgets(expr, sizeof(expr), stdin);
    expr[strcspn(expr, "\n")] = '\0'; // Remove newline

    pos = 0;
    if (E() && expr[pos] == '\0')
        printf("Valid arithmetic expression\n");
    else
        printf("Invalid arithmetic expression\n");

    return 0;
}

```
